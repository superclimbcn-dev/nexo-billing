export async function createServerClient() {
  const cookieStore = await cookies()

  return _createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return cookieStore.getAll() },
        setAll(toSet: { name: string; value: string; options: CookieOptions }[]) {
          try {
            toSet.forEach(({ name, value, options }) => cookieStore.set(name, value, options))

            // Called from a Server Component — cookies are read-only
          }
        },
      },
    }
  )

## 7. Middleware

**Archivo:** `apps/web/src/middleware.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { refreshSession } from '@nexo/core-auth'

const PUBLIC_PATHS = [
  '/login',
  '/signup',
  '/auth',
  '/check-email',
  '/auth-error',
  '/spoiler',
  '/api/health',
]

function isPublic(pathname: string) {
  return PUBLIC_PATHS.some((p) => pathname === p || pathname.startsWith(p + '/'))
}

export async function middleware(request: NextRequest) {
  const response = NextResponse.next()
  const { user } = await refreshSession(request, response)
  const { pathname } = request.nextUrl

  if (isPublic(pathname)) return response

  if (!user) {
    const loginUrl = new URL('/login', request.url)
    loginUrl.searchParams.set('next', pathname)
    return NextResponse.redirect(loginUrl)
  }

  const onboardingComplete = user.user_metadata?.onboarding_complete === true

  if (!onboardingComplete && !pathname.startsWith('/onboarding')) {
    return NextResponse.redirect(new URL('/onboarding/cuenta', request.url))
  }

  if (onboardingComplete && pathname.startsWith('/onboarding')) {
    return NextResponse.redirect(new URL('/dashboard', request.url))
  }

  return response
}

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico|public/).*)'],
}
```

## 10. RLS Policies activas

**Directorio:** `infrastructure/supabase/migrations/`

```

0002_rls_policies.sql         # Políticas SELECT/INSERT/UPDATE/DELETE por tenant_id

0004_hash_chain_constraint.sql # Constraint de encadenamiento hash en invoice_records
0005_utility_functions.sql    # Funciones PL/pgSQL auxiliares
0006_jwt_claims_hook.sql      # Custom Access Token Hook (promueve tenant_id + user_role al JWT)
0007_invitations_rls.sql      # RLS para tabla invitations (creada por Prisma en 20260425120000)
```

**Contenido de `0002_rls_policies.sql` (el más relevante):**

Tablas con RLS: `tenants`, `users`, `clients`, `products`, `invoices`, `invoice_lines`, `invoice_records`, `recurring_contracts`.

Patrón general: `tenant_id = (auth.jwt() ->> 'tenant_id')::uuid`

Excepción `tenants`: usa subquery `id = (SELECT tenant_id FROM users WHERE id = auth.uid())`.

Excepción `invoice_lines`: sin `tenant_id` directo — usa `EXISTS (SELECT 1 FROM invoices WHERE id = invoice_lines.invoice_id AND tenant_id = (auth.jwt() ->> 'tenant_id')::uuid)`.

**`invoice_records`:** solo SELECT para usuarios autenticados; INSERT/UPDATE requieren service role (admin client en app layer).

**`invitations` (`0007`):** SELECT para miembros del tenant, DELETE solo para OWNER/ADMIN, INSERT sin política de usuario (solo service role).

**`0006_jwt_claims_hook.sql` — Custom Access Token Hook:**

```sql
CREATE OR REPLACE FUNCTION public.custom_access_token_hook(event jsonb)
RETURNS jsonb LANGUAGE plpgsql STABLE SECURITY DEFINER AS $$
DECLARE
  claims jsonb;
  user_record record;
BEGIN
  claims := event -> 'claims';
  SELECT tenant_id, role INTO user_record
  FROM public.users WHERE id = (event ->> 'user_id')::uuid;

  IF FOUND THEN
    -- Uses "user_role" not "role" to avoid collision with Supabase's internal claim
    claims := jsonb_set(claims, '{tenant_id}', to_jsonb(user_record.tenant_id::text));
    claims := jsonb_set(claims, '{user_role}', to_jsonb(user_record.role::text));
  END IF;

  RETURN jsonb_set(event, '{claims}', claims);
END;
$$;
GRANT EXECUTE ON FUNCTION public.custom_access_token_hook TO supabase_auth_admin;
REVOKE EXECUTE ON FUNCTION public.custom_access_token_hook FROM authenticated, anon, public;
```

**TRAMPA CONOCIDA:** El hook debe activarse MANUALMENTE en el dashboard de Supabase (Authentication → Hooks). No es automático tras aplicar la migración.

## 11. Onboarding flow

**Rutas:** `app/(onboarding)/onboarding/`
```
cuenta/page.tsx         # Paso 1: nombre + email de acceso (magic link)
vertical/page.tsx       # Paso 2: selección de vertical (limpieza, construcción, etc.)
empresa/page.tsx        # Paso 3: NIF, razón social, dirección
datos-fiscales/page.tsx # Paso 4: régimen IVA, tipo IVA por defecto, año fiscal
configuracion/page.tsx  # Paso 5: email notificaciones + aceptar T&C → completeOnboarding()
```

**Estado del onboarding:** acumulado en `user_metadata.onboarding_state` (JSON en Supabase Auth) via `setOnboardingState()` Server Action.

**Server Action de creación atómica `completeOnboarding()`** — `apps/web/src/actions/onboarding.ts`:

```typescript
'use server'
import { randomUUID } from 'crypto'
import { redirect } from 'next/navigation'
import { createServerClient, createAdminClient } from '@nexo/core-auth'
import { prisma, UserRole } from '@nexo/prisma'

export async function completeOnboarding(formData: FormData) {
  // 1. Auth check
  const supabase = await createServerClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) redirect('/login')

  // 2. Recover accumulated onboarding state from user_metadata
  const state = (user.user_metadata?.onboarding_state as Record<string, unknown>) ?? {}
  const data = { regimenIva, ivaDefault, anioFiscal, emailNotificaciones } // from formData

  // 3. Pre-generate tenant ID (required for batch $transaction — see note below)
  const tenantId = randomUUID()

  // 4. Atomic batch transaction (array form, NOT callback form)
  //    CRITICAL: pgbouncer in transaction mode on Vercel serverless is incompatible
  //    with the callback form (prisma.$transaction(async tx => ...)) because it uses
  //    advisory locks. The array/batch form has no such requirement.
  try {
    await prisma.$transaction([
      prisma.tenant.create({ data: { id: tenantId, name, nif, vertical, plan: 'free', sectorMetadata: {...} } }),
      prisma.user.create({ data: { id: user.id, tenantId, email, name, role: UserRole.OWNER } }),
    ])
  } catch (err) {
    redirect(`/onboarding/configuracion?error=${encodeURIComponent(`Error al crear la cuenta: ${err.message}`)}`)
  }

  // 5. Promote tenant_id + role to app_metadata (consumed by JWT hook → RLS)
  const adminClient = createAdminClient()
  await adminClient.auth.admin.updateUserById(user.id, {
    app_metadata: { tenant_id: tenantId, role: UserRole.OWNER },
  })

  // 6. Mark onboarding complete
  await supabase.auth.updateUser({ data: { onboarding_complete: true, onboarding_state: null } })

  redirect('/dashboard')
}
```

## 15. Notas adicionales — trampas y deuda técnica conocida

### Decisiones arquitectónicas críticas

1. **pgbouncer + Prisma en Vercel serverless**: `prisma.$transaction(async tx => ...)` (forma callback) es INCOMPATIBLE con pgbouncer en modo transaction porque usa advisory locks. Toda transacción Prisma en este proyecto DEBE usar la forma array: `prisma.$transaction([op1, op2])`. Los IDs que una operación necesita pasar a la siguiente deben pre-generarse con `randomUUID()`.

2. **Prisma engine binary en Vercel**: `next.config.ts` usa `outputFileTracingIncludes` con patrón `/**/*` (NO `/api/**/*`). Si se cambia a un patrón más restrictivo, el binario `rhel-openssl-3.0.x` dejará de incluirse en los Lambdas de rutas no-API (layouts, pages), causando `PrismaClientInitializationError` en producción.

3. **JWT hook manual**: La función `custom_access_token_hook` se aplica vía migración SQL pero debe activarse **manualmente** en Supabase dashboard → Authentication → Hooks. Sin esto, `(auth.jwt() ->> 'tenant_id')` devuelve NULL y RLS bloquea todo.

4. **`user_role` no `role` en JWT**: El hook inyecta `user_role` (no `role`) para evitar colisión con el claim interno de Supabase. Las RLS policies de `invitations` usan `auth.jwt() ->> 'user_role'`. Si se crean nuevas políticas, deben usar `user_role` no `role`.

5. **`app_metadata` vs JWT para tenant_id**: El middleware y los Server Components leen `tenant_id` de `user.app_metadata` (disponible inmediatamente tras `updateUserById`). Las RLS policies leen de `auth.jwt() ->> 'tenant_id'` (disponible en el siguiente refresh de token). Son dos rutas de lectura distintas — no intercambiables.

### Deuda técnica

- **Cero tests**: No hay ningún test unitario ni E2E. El módulo Verifactu (cuando se implemente) requiere 90% de cobertura por CLAUDE.md.
- **Sin i18n formal**: Los strings de UI están hardcodeados en español. CLAUDE.md menciona `packages/core-ui/i18n` pero no existe.
- **`core-types` vacío**: Zod schemas compartidos están pendientes de creación.
- **Stubs sin estructura**: `core-billing`, `core-crm`, `core-analytics`, `core-payments` y todos los verticales son carpetas vacías sin `package.json` funcional, `tsconfig.json` ni `src/index.ts`.
- **`settings/` incompleta**: Solo existe `settings/team`. No hay `settings/` raíz, ni branding, ni billing.
- **Flujo de aceptación de invitación sin verificar**: La invitación se crea y el email se envía via Supabase, pero el flujo de aceptación (magic link → crear `public.users` + actualizar `app_metadata`) no ha sido validado end-to-end en staging/prod por rate-limit de Supabase durante el desarrollo.
- **`app/page.tsx` y `app/layout.tsx` raíz**: No documentados en este snapshot — probablemente una redirección a `/dashboard` o `/login`.
